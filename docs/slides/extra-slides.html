DAG

library(dagitty)
library(ggdag)

# coordinates for graph
coords <- read.table(header = TRUE, text = "
  x y name
  0 -1 U
  0 1 L
  2 0 A0
  3 0 Y0
  4 0 A1
  6 0 Y2
")

# make into daggity object
d <- dagify(
  L ~ U,
  A0 ~  L + U,
  Y0 ~  L + U,
  A1 ~  L + U ,
  Y2 ~  L + U,
  exposure = "A1",
  outcome = "Y2",
  latent = "U",
  coords = coords)

# plot
ggdag(d,  text = TRUE, node = FALSE, 
      text_size = 9,
      text_col = "black") +
   theme_dag()
